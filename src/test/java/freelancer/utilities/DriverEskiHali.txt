  private Driver() {
    }

    static WebDriver driver;

    public static WebDriver getDriver() {

        if (driver == null) {

            driver = new ChromeDriver();
            driver.manage().window().maximize();
            driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(20));
        }
        return driver;
    }
